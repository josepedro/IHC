  \fi%

  \ifnum\alphainsection=1% 
    \Alph{alphasect}
  \else%
    \arabic{section}
  \fi%

  \ifnum\alphainsection=1%

  \fi%
  \setcounter{alphasect}{0}
  \def\alphainsection{1}

  \setcounter{alphasect}{0}
  \def\alphainsection{0}
